BEGIN
    READLN(n);
    a := 1;
    b := 1;
    c := a + b;
    IF n = 1 THEN WRITELN(a);
    IF n = 2 THEN WRITELN(b);
    IF n > 2 THEN BEGIN
        WHILE n > 2 DO BEGIN
            c := a + b;
            a := b;
            b := c;
            n := n - 1;
        END;
        WRITELN(c);
    END;
END;
